021package org.codehaus.groovy.grails.plugins.gwt;
022
023import org.codehaus.groovy.grails.commons.ArtefactHandlerAdapter;
024
025/**
026 * Manages the GWT action handler artifact type. Action handlers are
027 * the server-side components that process the actions (or commands)
028 * that come from the GWT client. Actions are equivalent to commands
029 * in the Command pattern.
030 */
031public final class ActionHandlerArtefactHandler extends ArtefactHandlerAdapter {
032    public static final String TYPE = "ActionHandler";
033
034    public ActionHandlerArtefactHandler() {
035        super(TYPE, ActionHandlerGrailsClass.class, DefaultActionHandlerGrailsClass.class, null);
036    }
037
038    public boolean isArtefactClass(Class clazz) {
039        return clazz != null && clazz.getName().endsWith(TYPE);
040    }
041}
