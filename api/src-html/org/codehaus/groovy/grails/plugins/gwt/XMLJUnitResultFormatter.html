056        private static final double ONE_SECOND = 1000.0;
057        
058        /** constant for unnnamed testsuites/cases */
059        private static final String UNKNOWN = "unknown";
060    
061        private static final String GWT_PRODUCTION_PREFIX = "GWT_PROD_MODE-";
062        
063        private static DocumentBuilder getDocumentBuilder() {
064            try {
065                return DocumentBuilderFactory.newInstance().newDocumentBuilder();
066            } catch (Exception exc) {
067                throw new ExceptionInInitializerError(exc);
068            }
069        }

071        /**
072         * The XML document.
073         */
074        private Document doc;
075        /**
076         * The wrapper for the whole testsuite.
077         */
078        private Element rootElement;
079        /**
080         * Element for the current test.
081         */
082        private Hashtable testElements = new Hashtable();
083        /**
084         * tests that failed.
085         */
086        private Hashtable failedTests = new Hashtable();
087        /**
088         * Timing helper.
089         */
090        private Hashtable testStarts = new Hashtable();
091        /**
092         * Where to write the log to.
093         */
094        private OutputStream out;
095    
096        private boolean gwtProductionMode;

098        /** No arg constructor. */
099        public XMLJUnitResultFormatter() {
100            // check if we run in gwt production mode
101            String gwtArgs = System.getProperty("gwt.args");
102            gwtProductionMode = gwtArgs != null && gwtArgs.indexOf("-prod") != -1;
103        }

120        /**
121         * The whole testsuite started.
122         * @param suite the testsuite.
123         */
124        public void startTestSuite(JUnitTest suite) {
125            doc = getDocumentBuilder().newDocument();
126            rootElement = doc.createElement(TESTSUITE);
127            String n = suite.getName();
128            rootElement.setAttribute(ATTR_NAME, n == null ? UNKNOWN : n);
129    
130            //add the timestamp
131            final String timestamp = DateUtils.format(new Date(),
132                    DateUtils.ISO8601_DATETIME_PATTERN);
133            rootElement.setAttribute(TIMESTAMP, timestamp);
134            //and the hostname.
135            rootElement.setAttribute(HOSTNAME, getHostname());
136    
137            // Output properties
138            Element propsElement = doc.createElement(PROPERTIES);
139            rootElement.appendChild(propsElement);
140            Properties props = suite.getProperties();
141            if (props != null) {
142                Enumeration e = props.propertyNames();
143                while (e.hasMoreElements()) {
144                    String name = (String) e.nextElement();
145                    Element propElement = doc.createElement(PROPERTY);
146                    propElement.setAttribute(ATTR_NAME, name);
147                    propElement.setAttribute(ATTR_VALUE, props.getProperty(name));
148                    propsElement.appendChild(propElement);
149                }
150            }
151        }
152    
153        /**
154         * get the local hostname
155         * @return the name of the local host, or "localhost" if we cannot work it out
156         */
157        private String getHostname()  {
158            try {
159                return InetAddress.getLocalHost().getHostName();
160            } catch (UnknownHostException e) {
161                return "localhost";
162            }
163        }
164    
165        /**
166         * The whole testsuite ended.
167         * @param suite the testsuite.
168         * @throws BuildException on error.
169         */
170        public void endTestSuite(JUnitTest suite) throws BuildException {
171            rootElement.setAttribute(ATTR_TESTS, "" + suite.runCount());
172            rootElement.setAttribute(ATTR_FAILURES, "" + suite.failureCount());
173            rootElement.setAttribute(ATTR_ERRORS, "" + suite.errorCount());
174            rootElement.setAttribute(
175                ATTR_TIME, "" + (suite.getRunTime() / ONE_SECOND));
176            if (out != null) {
177                Writer wri = null;
178                try {
179                    wri = new BufferedWriter(new OutputStreamWriter(out, "UTF8"));
180                    wri.write("<?xml version=\"1.0\" encoding=\"UTF-8\" ?>\n");
181                    (new DOMElementWriter()).write(rootElement, wri, 0, "  ");
182                    wri.flush();
183                } catch (IOException exc) {
184                    throw new BuildException("Unable to write log file", exc);
185                } finally {
186                    if (out != System.out && out != System.err) {
187                        FileUtils.close(wri);
188                    }
189                }
190            }
191        }

193        /**
194         * Interface TestListener.
195         *
196         * <p>A new Test is started.
197         * @param t the test.
198         */
199        public void startTest(Test t) {
200            testStarts.put(t, new Long(System.currentTimeMillis()));
201        }
202    
203        /**
204         * Interface TestListener.
205         *
206         * <p>A Test is finished.
207         * @param test the test.
208         */
209        public void endTest(Test test) {
210            // Fix for bug #5637 - if a junit.extensions.TestSetup is
211            // used and throws an exception during setUp then startTest
212            // would never have been called
213            if (!testStarts.containsKey(test)) {
214                startTest(test);
215            }
216    
217            Element currentTest = null;
218            if (!failedTests.containsKey(test)) {
219                currentTest = doc.createElement(TESTCASE);
220                String n = JUnitVersionHelper.getTestCaseName(test);
221                currentTest.setAttribute(ATTR_NAME,
222                                         n == null ? UNKNOWN : n);
223                // a TestSuite can contain Tests from multiple classes,
224                // even tests with the same name - disambiguate them.
225                String className = gwtProductionMode ? GWT_PRODUCTION_PREFIX+JUnitVersionHelper.getTestCaseClassName(test) : JUnitVersionHelper.getTestCaseClassName(test);
226                currentTest.setAttribute(ATTR_CLASSNAME, className);
227                rootElement.appendChild(currentTest);
228                testElements.put(test, currentTest);
229            } else {
230                currentTest = (Element) testElements.get(test);
231            }
232    
233            Long l = (Long) testStarts.get(test);
234            currentTest.setAttribute(ATTR_TIME,
235                "" + ((System.currentTimeMillis()
236                       - l.longValue()) / ONE_SECOND));
237        }

239        /**
240         * Interface TestListener for JUnit &lt;= 3.4.
241         *
242         * <p>A Test failed.
243         * @param test the test.
244         * @param t the exception.
245         */
246        public void addFailure(Test test, Throwable t) {
247            formatError(FAILURE, test, t);
248        }
249    
250        /**
251         * Interface TestListener for JUnit &gt; 3.4.
252         *
253         * <p>A Test failed.
254         * @param test the test.
255         * @param t the assertion.
256         */
257        public void addFailure(Test test, AssertionFailedError t) {
258            addFailure(test, (Throwable) t);
259        }
260    
261        /**
262         * Interface TestListener.
263         *
264         * <p>An error occurred while running the test.
265         * @param test the test.
266         * @param t the error.
267         */
268        public void addError(Test test, Throwable t) {
269            formatError(ERROR, test, t);
270        }
271    
272        private void formatError(String type, Test test, Throwable t) {
273            if (test != null) {
274                endTest(test);
275                failedTests.put(test, test);
276            }
277    
278            Element nested = doc.createElement(type);
279            Element currentTest = null;
280            if (test != null) {
281                currentTest = (Element) testElements.get(test);
282            } else {
283                currentTest = rootElement;
284            }
285    
286            currentTest.appendChild(nested);
287    
288            String message = t.getMessage();
289            if (message != null && message.length() > 0) {
290                nested.setAttribute(ATTR_MESSAGE, t.getMessage());
291            }
292            nested.setAttribute(ATTR_TYPE, t.getClass().getName());
293    
294            String strace = JUnitTestRunner.getFilteredTrace(t);
295            Text trace = doc.createTextNode(strace);
296            nested.appendChild(trace);
297        }
298    
299        private void formatOutput(String type, String output) {
300            Element nested = doc.createElement(type);
301            rootElement.appendChild(nested);
302            nested.appendChild(doc.createCDATASection(output));
303        }
